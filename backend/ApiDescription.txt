"/whatsUp" 
Request model:
{
	userId - int.
}
JSON Request body examples:
{"userId":144}
Returns object: 
{
    state - string enum { "WaitingForStart", "CreatingFleet", "YourTurn", "OpponentsTurn" }
	userId - int optional
}


"/createFleet"
Request model:
{
	userId - integer.
	ships - array of objects:
	{
		decks - integer array.
	}
}
Example request body JSON:
{
	"userId":144,
	"ships":
	[{
		"decks":[1,2,3]
	}]
}

"/attack"
Request model:
{
	userId - integer.
	location - integer.
}
Returns object:
{
	result - enum string { "Hit","Killed","Missed","Win" },
	fleet1 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			location - integer.
			destroyed - boolean.
		}
	}
	fleet2 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			location - integer.
			destroyed - boolean.
		}
	}
}
