"/whatsUp" 
Request model:
{
	userId - int.
}
Returns string enum { "WaitingForStart", "CreatingFleet", "YourTurn", "OpponentsTurn" }.

"/createFleet"
Request model:
{
	userId - integer,
	ships - array of objects:
	{
		decks - array of objects:
		{
			x - integer,
			y - integer.
		}
	}
}
Returns nothing.

"/attack"
Request model:
{
	userId - integer,
	x - integer,
	y - integer.
}
Returns object:
{
	result - enum string { "Hit","Killed","Missed","Win" },
	fleet1 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
	fleet2 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
}
