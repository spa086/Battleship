"/whatsUp" 
Request object:
{
	userId - int.
}
Returns object:
{
	gameState - string enum { "WaitingForStart", "CreatingFleet", "YourTurn", "OpponentsTurn" }
	fleet1 (optional) - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
	fleet2 (optional) - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
}

"/createFleet"
Request object:
{
	userId - integer,
	ships - array of objects:
	{
		decks - array of objects:
		{
			x - integer,
			y - integer.
		}
	}
}
Returns nothing.

"/attack"
Request model:
{
	userId - integer,
	location - object:
	{
		x - integer,
		y - integer.
	}
}
Returns object:
{
	result - enum string { "Hit","Killed","Missed","Win" },
	fleet1 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
	fleet2 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
}
