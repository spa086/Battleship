"/whatsUp" 
Request model:
{
	userId - int.
}
Returns object: 
{
    state - string enum { "WaitingForStart", "CreatingFleet", "YourTurn", "OpponentsTurn" }
	userId - int optional
}


"/createFleet"
Request model:
{
	userId - integer.
	ships - array of objects:
	{
		decks - array of objects:
		{
			"x" - integer,
			"y" - integer.
		}
	}
}

"/attack"
Request model:
{
	userId - integer,
	x - integer,
	y - integer.
}
Returns object:
{
	result - enum string { "Hit","Killed","Missed","Win" },
	fleet1 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
	fleet2 - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
}
