"/whatsUp" 
Request object:
{
	userId - int.
}
Returns object:
{
	gameState - string enum { "WaitingForStart", "CreatingFleet", "YourTurn", "OpponentsTurn", "YouWon", "OpponentWon" }
	myFleet (optional) - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
	opponentFleet (optional) - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	},
	myExcludedLocations - array of objects:
	{
		x - integer,
		y - integer.
	},
	opponentExcludedLocations - array of objects:
	{
		x - integer,
		y - integer.
	},
	gameId - integer.
}

"/createFleet"
Request object:
{
	userId - integer,
	ships - array of objects:
	{
		decks - array of objects:
		{
			x - integer,
			y - integer.
		}
	}
}
Returns nothing.

"/attack"
Request object:
{
	userId - integer,
	location - object:
	{
		x - integer,
		y - integer.
	}
}
Returns object:
{
	result - string enum { "Hit","Killed","Missed","Win" },
	myFleet - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	}
	opponentFleet - array of objects:
	{
		userId - integer.
		decks - array of objects: 
		{
			x - integer,
			y - integer,
			destroyed - boolean.
		}
	},
	myExcludedLocations - array of objects:
	{
		x - integer,
		y - integer.
	},
	opponentExcludedLocations - array of objects:
	{
		x - integer,
		y - integer.
	}
}
